public class practice4{

    static int factorial(int n){

        int result=1;

        for (int i=1; i<=n; i++){
            result=result*i;
        }
        return result;
    }

    static long factorialL(int n){

        long result=1L;

        for (int i=1; i<=n; i++){
            result=result*i;
        }
        return result;
    }

    static double factorialD(int n){

        double result=1.0;

        for (int i=1; i<=n; i++){
            result=result*i;
        }
        return result;
    }

        double sum=1.0;
        double term=1.0;

            k++;
        }

        return sum;
    }

    }
